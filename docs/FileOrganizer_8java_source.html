53 public void organizeFiles(Path folder, Consumer<Double> progressConsumer) {

58 Thread progressThread = new Thread(() -> ProgressUtils.simulateProgress(progressConsumer, completed));
59 progressThread.setDaemon(true); // Thread daemon (no bloqueja tancament de l'aplicació)
60 progressThread.start();
61
63 new Thread(() -> {
65 try (Stream<Path> files = Files.list(folder)) { // Llista tots els fitxers de la carpeta
67 Map<String, Integer> categoryCount = new HashMap<>();
68
70 files.forEach(file -> {
71 if (Files.isRegularFile(file)) { // Comprova que sigui un fitxer regular
72 try {
74 String extension = getExtension(file.getFileName().toString());
75 String category = getCategoryForExtension(extension);
76 Path targetDir = folder.resolve(category); // Crea la ruta destí
77
79 Files.createDirectories(targetDir);
81 Files.move(file, targetDir.resolve(file.getFileName()));
82
84 categoryCount.merge(category, 1, Integer::sum);
85 } catch (IOException e) {
87 logger.log("❌ Error movent fitxer: " + e.getMessage());
88 }
89 }
90 });
91
93 logger.log("✔️ Organització completa");
95 categoryCount.forEach((category, count) ->
96 logger.log(" " + category + ": " + count + " fitxers"));
97 } catch (IOException e) {
99 logger.log("❌ Error organitzant: " + e.getMessage());
100 } finally {
102 completed.set(true);
103 }
104 }).start(); // Inicia el thread d'organització
105 }

116 private String getExtension(String fileName) {

132 private String getCategoryForExtension(String extension) {
134 if (extension.matches("zip|rar|7z|tar|gz|bz2")) return "comprimits";
135 if (extension.matches("jpg|jpeg|png|gif|bmp|webp|svg")) return "imatges";
136 if (extension.matches("mp3|wav|ogg|m4a|flac")) return "audio";
137 if (extension.matches("mp4|avi|mkv|mov|wmv")) return "video";
138 if (extension.matches("doc|docx|pdf|txt|rtf|odt")) return "documents";
139 if (extension.matches("exe|msi|deb|rpm|appimage")) return "executables";
140 if (extension.matches("js|ts|java|py|cpp|cs|php|html|css")) return "codi";
141 if (extension.matches("json|xml|yaml|yml|ini|conf")) return "config";
142 return "others"; // Categoria per defecte
143 }
